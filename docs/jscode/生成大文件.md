### 大文件

```js
var fs = require('fs')

let file = fs.createWriteStream('test2.txt',{flags: 'a'})

let str = ''

for(let i =0 ;i < 1024 ; i++) {
  str += '0'
}
for (let i = 0; i < 1024 * 1024 * 2; i++) {
  file.write(str)
}



file.end('end!', () => {
  console.log('end!')
})

```
